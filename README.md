Algoritmo "Revisão Laço de Repetição"

Var

   numero, i : inteiro

Inicio

   escreva("Digite um número: ")
   leia(numero)

   escreva("Digite um número: ")
   leia(numero)

   escreva("Digite um número: ")
   leia(numero)

   escreva("Digite um número: ")
   leia(numero)

   escreva("Digite um número: ")
   leia(numero)

   para i de 1 ate 5 faca
      escreva("Digite um número: ")
      leia(numero)
   fimpara
